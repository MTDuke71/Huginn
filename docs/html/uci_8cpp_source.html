246void UCIInterface::handle_position(const std::vector<std::string>& tokens) {
247 if (tokens.size() < 2) return;
248
249 size_t move_index = 0;
250
251 if (tokens[1] == "startpos") {
252 position.set_startpos();
253 move_index = 2;
254 }
255 else if (tokens[1] == "fen") {
256 if (tokens.size() < 3) return; // Need at least "position fen [something]"
257
258 // Build FEN string from tokens until we find "moves" or reach end
259 std::string fen;
260 size_t i = 2;
261 while (i < tokens.size() && tokens[i] != "moves") {
262 if (i > 2) fen += ' ';
263 fen += tokens[i];
264 i++;
265 }
266
267 if (fen.empty()) {
268 if (debug_mode) std::cout << "info string Empty FEN string" << std::endl;
269 return;
270 }
271
272 if (!position.set_from_fen(fen)) {
273 if (debug_mode) std::cout << "info string Invalid FEN: " << fen << std::endl;
274 return;
275 }
276
277 move_index = i; // Points to "moves" or end of tokens
278 }
279 else {
280 if (debug_mode) std::cout << "info string Unknown position type: " << tokens[1] << std::endl;
281 return;
282 }
283
284 // Process moves if present
285 if (move_index < tokens.size() && tokens[move_index] == "moves") {
286 for (size_t i = move_index + 1; i < tokens.size(); ++i) {
287 S_MOVE move = parse_uci_move(tokens[i], position);
288 if (move.move != 0) {
289 if (position.MakeMove(move) != 1) {
290 if (debug_mode) std::cout << "info string Illegal move: " << tokens[i] << std::endl;
291 }
292 } else if (debug_mode) {
293 std::cout << "info string Invalid move: " << tokens[i] << std::endl;
294 }
295 }
296 }
297
298 if (debug_mode) std::cout << "info string Position set, FEN: " << position.to_fen() << std::endl;
299}

313void UCIInterface::handle_go(const std::vector<std::string>& tokens) {
314 if (debug_mode) std::cout << "info string Starting search" << std::endl;
315 should_stop = false;
316
317 Huginn::MinimalLimits limits;
318 limits.infinite = false;

347 } catch (const std::exception&) {

350 }
351 }
352 else if (tokens[i] == "wtime" && i + 1 < tokens.size()) {

389 i++;

392 else if (tokens[i] == "movestogo" && i + 1 < tokens.size()) {
393 try {
394 movestogo = std::stoi(tokens[i + 1]);
395 if (debug_mode) std::cout << "info string Parsed movestogo: " << movestogo << std::endl;
396 i++;
397 } catch (const std::exception&) {
398 if (debug_mode) std::cout << "info string Invalid movestogo value: " << tokens[i + 1] << std::endl;
399 i++;
400 }
401 }
402 else if (tokens[i] == "infinite") {
403 limits.infinite = true;
404 limits.max_time_ms = 0;
405 if (debug_mode) std::cout << "info string Parsed infinite search mode" << std::endl;
406 }
407 else if (debug_mode) {
408 std::cout << "info string Unknown go parameter: " << tokens[i] << std::endl;
409 }
410 }
411
412 // VICE Part 69: Set search mode based on parsed parameters
413 if (depth_specified) {
414 limits.infinite = true; // Depth-only search ignores time
415 limits.max_time_ms = 0;
416 if (debug_mode) std::cout << "info string Using depth-only search: " << limits.max_depth << std::endl;
417 }
418 else if (movetime > 0) {
419 // Fixed time per move
420 if (debug_mode) std::cout << "info string Using fixed movetime: " << movetime << "ms" << std::endl;
421 }
422 else if (!limits.infinite && (wtime > 0 || btime > 0)) {
423 // Time control calculation (VICE Part 69 style with our enhancements)
424 int side_time = (position.side_to_move == Color::White) ? wtime : btime;
425 int side_inc = (position.side_to_move == Color::White) ? winc : binc;
426
427 if (side_time > 0) {
428 int alloc;
429 if (movestogo > 0) {
430 // Classical time control: divide remaining time by moves to go
431 alloc = side_time / std::max(1, movestogo);
432 alloc += side_inc / 2; // Add half the increment
433 if (debug_mode) std::cout << "info string Classical time control: " << side_time << "ms / " << movestogo << " moves" << std::endl;
434 } else {
435 // Sudden death or increment-only time control
436 alloc = std::max(50, side_time / 20 + side_inc / 4);
437 if (debug_mode) std::cout << "info string Sudden death time control: " << side_time << "ms base" << std::endl;
438 }
439
440 // Apply reserve time (don't use all available time)
441 int reserve = std::min(1000, std::max(50, side_time / 10));
442 alloc = std::max(50, alloc - reserve / std::max(1, movestogo > 0 ? movestogo : 1));
443
444 // Cap at 60% of remaining time for safety
445 alloc = std::min(alloc, static_cast<int>(side_time * 0.6));
446
447 limits.max_time_ms = std::max(50, alloc);
448 if (debug_mode) std::cout << "info string Calculated time allocation: " << limits.max_time_ms << "ms (reserve: " << reserve << "ms)" << std::endl;
449 } else {
450 // No time available, use increment if present
451 if (movestogo == 0 && (winc > 0 || binc > 0)) {
452 limits.max_time_ms += ((position.side_to_move == Color::White) ? winc : binc) / 4;
453 if (debug_mode) std::cout << "info string Using increment-only time: " << limits.max_time_ms << "ms" << std::endl;
454 }
455 }
456 }
457
458 if (debug_mode) {
459 std::cout << "info string Final search parameters: depth=" << limits.max_depth
460 << " time=" << limits.max_time_ms << "ms infinite=" << (limits.infinite ? "true" : "false") << std::endl;
461 }
462
463 search_best_move(limits);
464}

477void UCIInterface::handle_setoption(const std::vector<std::string>& tokens) {
478 if (tokens.size() < 4) return; // Need at least "setoption name X value Y"

480 if (tokens[1] != "name") return;

482 std::string option_name = tokens[2];
483
484 if (tokens.size() >= 5 && tokens[3] == "value") {
485 std::string option_value = tokens[4];
486
487 if (option_name == "Hash") {

510 bool new_own_book = (option_value == "true");
511 if (new_own_book != own_book) {
512 own_book = new_own_book;
513 if (own_book) {
