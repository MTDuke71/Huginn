89inline bool sliding_attacks_rank_file(int piece_sq, int target_sq, const Position& pos) {
90 // Check if on same rank or file
91 File piece_file = file_of(piece_sq);
92 Rank piece_rank = rank_of(piece_sq);
93 File target_file = file_of(target_sq);
94 Rank target_rank = rank_of(target_sq);
95
96 if (piece_file != target_file && piece_rank != target_rank) {
97 return false; // Not on same rank or file
98 }
99
100 // Determine direction
101 int dir;
102 if (piece_file == target_file) {
103 // Same file - moving along rank
104 dir = (target_rank > piece_rank) ? NORTH : SOUTH;
105 } else {
106 // Same rank - moving along file
107 dir = (target_file > piece_file) ? EAST : WEST;
108 }
109
110 // Check for clear path
111 int current_sq = piece_sq + dir;
112 while (current_sq != target_sq && is_playable(current_sq)) {
113 if (!is_none(pos.at(current_sq))) {
114 return false; // Path blocked
115 }
116 current_sq += dir;
117 }
118
119 return current_sq == target_sq;
120}

135inline bool sliding_attacks_diagonal(int piece_sq, int target_sq, const Position& pos) {
136 // Check if on same diagonal
137 int file_diff = int(file_of(target_sq)) - int(file_of(piece_sq));
138 int rank_diff = int(rank_of(target_sq)) - int(rank_of(piece_sq));
139
140 if (abs(file_diff) != abs(rank_diff)) {
141 return false; // Not on same diagonal
142 }
143
144 // Determine direction
145 int dir;
146 if (file_diff > 0 && rank_diff > 0) dir = NE;
147 else if (file_diff < 0 && rank_diff > 0) dir = NW;
148 else if (file_diff > 0 && rank_diff < 0) dir = SE;
149 else dir = SW;
150
151 // Check for clear path
152 int current_sq = piece_sq + dir;
153 while (current_sq != target_sq && is_playable(current_sq)) {
154 if (!is_none(pos.at(current_sq))) {
155 return false; // Path blocked

157 current_sq += dir;
158 }
