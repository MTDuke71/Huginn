97void generate_pawn_moves(const Position& pos, S_MOVELIST& list, Color us) {
98 int piece_count = pos.pCount[int(us)][int(PieceType::Pawn)];
99
100 for (int i = 0; i < piece_count; ++i) {
101 int from = pos.pList[int(us)][int(PieceType::Pawn)][i];
102 if (from == -1) continue; // Safety check
103
104 // Determine pawn direction and starting rank
105 int direction = (us == Color::White) ? NORTH : SOUTH;
106 Rank start_rank = (us == Color::White) ? Rank::R2 : Rank::R7;
107 Rank promo_rank = (us == Color::White) ? Rank::R8 : Rank::R1;
108
109 // Single push
110 int to = from + direction;
111 if (IS_PLAYABLE(to) && pos.at(to) == Piece::None) {
112 if (rank_of(to) == promo_rank) {
113 // Promotions
114 list.add_promotion_move(make_promotion(from, to, PieceType::Queen));
115 list.add_promotion_move(make_promotion(from, to, PieceType::Rook));
116 list.add_promotion_move(make_promotion(from, to, PieceType::Bishop));
117 list.add_promotion_move(make_promotion(from, to, PieceType::Knight));
118 } else {
119 list.add_quiet_move(make_move(from, to));
120
121 // Double push from starting position
122 if (rank_of(from) == start_rank) {
123 int double_to = to + direction;
124 if (IS_PLAYABLE(double_to) && pos.at(double_to) == Piece::None) {
125 list.add_quiet_move(make_pawn_start(from, double_to));
126 }
127 }
128 }
129 }
130
131 // Captures (diagonals)
132 for (int capture_dir : {direction + WEST, direction + EAST}) {
133 int capture_to = from + capture_dir;
134 if (IS_PLAYABLE(capture_to)) {
135 Piece target = pos.at(capture_to);
136
137 if (target != Piece::None && color_of(target) == !us) {
138 // Regular capture
139 if (rank_of(capture_to) == promo_rank) {
140 // Capture promotion
141 list.add_promotion_move(make_promotion(from, capture_to, PieceType::Queen, type_of(target)));
142 list.add_promotion_move(make_promotion(from, capture_to, PieceType::Rook, type_of(target)));
143 list.add_promotion_move(make_promotion(from, capture_to, PieceType::Bishop, type_of(target)));
144 list.add_promotion_move(make_promotion(from, capture_to, PieceType::Knight, type_of(target)));
145 } else {
146 list.add_capture_move(make_capture(from, capture_to, type_of(target)), pos);
147 }
148 } else if (capture_to == pos.ep_square) {
149 // En passant capture
150 list.add_en_passant_move(make_en_passant(from, capture_to));
151 }
152 }
153 }
154 }
155}

157void generate_knight_moves(const Position& pos, S_MOVELIST& list, Color us) {
158 int piece_count = pos.pCount[int(us)][int(PieceType::Knight)];
159
160 for (int i = 0; i < piece_count; ++i) {
161 int from = pos.pList[int(us)][int(PieceType::Knight)][i];
162 if (from == -1) continue;
163
164 for (int delta : KNIGHT_DELTAS) {
165 int to = from + delta;
166 if (IS_PLAYABLE(to)) {
167 Piece target = pos.at(to);
168
169 if (target == Piece::None) {
170 list.add_quiet_move(make_move(from, to));
171 } else if (color_of(target) == !us) {
172 list.add_capture_move(make_capture(from, to, type_of(target)), pos);
173 }

179void generate_sliding_moves(const Position& pos, S_MOVELIST& list, Color us, PieceType piece_type, const int* directions, int num_directions) {
180 int piece_count = pos.pCount[int(us)][int(piece_type)];
181
182 for (int i = 0; i < piece_count; ++i) {
183 int from = pos.pList[int(us)][int(piece_type)][i];
184 if (from == -1) continue;
185
186 for (int d = 0; d < num_directions; ++d) {
187 int direction = directions[d];
188
189 for (int to = from + direction; IS_PLAYABLE(to); to += direction) {
190 Piece target = pos.at(to);
191
192 if (target == Piece::None) {
193 list.add_quiet_move(make_move(from, to));
194 } else {
195 if (color_of(target) == !us) {
196 list.add_capture_move(make_capture(from, to, type_of(target)), pos);
197 }
198 break; // Piece blocks further movement
199 }
200 }
201 }

205// Enhanced legal move generation with better performance
206void generate_legal_moves_enhanced(Position& pos, S_MOVELIST& list) {
207 S_MOVELIST pseudo_moves;
208 generate_all_moves(pos, pseudo_moves);
209
210 list.count = 0; // Direct clear - faster than function call
211
212 // BUGFIX: Work on a copy to avoid modifying the input position
213 Position temp_pos = pos;
214
215 // Filter out illegal moves (those that leave king in check)
216 for (int i = 0; i < pseudo_moves.size(); ++i) {
217 if (temp_pos.MakeMove(pseudo_moves[i]) == 1) {
218 list.add_quiet_move(pseudo_moves[i]); // Move is already legal
219 temp_pos.TakeMove();
220 }
221 }
222}
