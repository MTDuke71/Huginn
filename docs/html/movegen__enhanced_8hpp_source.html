137// Check if a move is legal (doesn't leave own king in check)
138inline bool is_legal_move(Position& pos, const S_MOVE& move) {
139 // Special handling for castling
140 if (move.is_castle()) {
141 int from = move.get_from();
142 int to = move.get_to();
143 Color current_side = pos.side_to_move;
144 Color opponent_side = (current_side == Color::White) ? Color::Black : Color::White;
145
146 // King cannot be in check before castling
147 if (SqAttacked(from, pos, opponent_side)) {
148 return false;
149 }
150
151 // Check that king doesn't pass through attacked squares during castling
152 int step = (to > from) ? 1 : -1;
153 for (int sq = from + step; sq != to + step; sq += step) {
154 if (SqAttacked(sq, pos, opponent_side)) {
155 return false;
156 }
157 }
158 return true;
159 }
160
161 // For all other moves, use the proper move/undo system
162 Color current_side = pos.side_to_move;
163
164 // Apply the move using VICE method
165 if (pos.MakeMove(move) != 1) {
166 return false; // Move was illegal
167 }
168
169 // Get the king position after the move (for the side that just moved)
170 int king_sq = pos.king_sq[int(current_side)];
171
172 // Check if our king would be in check after the move
173 // Note: after the move, it's the opponent's turn, so we check if opponent attacks our king
174 bool legal = !SqAttacked(king_sq, pos, !current_side);
175
176 // Undo the move to restore the original position
177 pos.TakeMove();
178
179 return legal;
