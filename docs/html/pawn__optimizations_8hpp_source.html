64// Batch promotion move generation - generates all 4 promotion moves in tight loop
65// This replaces the individual add_promotion_move calls that were causing overhead
66inline void generate_promotion_batch(S_MOVELIST& list, int from, int to, PieceType captured = PieceType::None) {
67 // Pre-calculate common values to avoid repeated computation
68 const bool is_capture = (captured != PieceType::None);
69 const int base_score = 2000000; // Base promotion score
70
71 // Generate all 4 promotions in tight loop - compiler can optimize this better
72 for (int i = 0; i < 4; ++i) {
73 PieceType promoted = PROMOTION_PIECES[i];
74 S_MOVE move = make_promotion(from, to, promoted, captured);
75
76 // Optimized scoring calculation (avoiding function calls in hot path)
77 int score = base_score;
78
79 // Add promotion piece bonus (Queen=900, Rook=500, Bishop=350, Knight=300)
80 switch (promoted) {
81 case PieceType::Queen: score += 90000; break;
82 case PieceType::Rook: score += 50000; break;
83 case PieceType::Bishop: score += 35000; break;
84 case PieceType::Knight: score += 30000; break;
85 default: break;
86 }
87
88 // Add capture bonus if applicable
89 if (is_capture) {
90 switch (captured) {
91 case PieceType::Queen: score += 9000; break;
92 case PieceType::Rook: score += 5000; break;
93 case PieceType::Bishop: score += 3500; break;
94 case PieceType::Knight: score += 3000; break;
95 case PieceType::Pawn: score += 1000; break;
96 default: break;
97 }
98 }
99
100 // Add move directly to list (avoiding add_promotion_move overhead)
101 list.moves[list.count] = move;
102 list.moves[list.count].score = score;
103 list.count++;
104 }
105}

107// Specialized pawn move generation for promotion-heavy positions
108// Uses early promotion square detection to avoid repeated rank calculations
109inline void generate_pawn_moves_optimized(const Position& pos, S_MOVELIST& list, Color us) {
110 const int piece_count = pos.pCount[int(us)][int(PieceType::Pawn)];
111
112 // Early exit for positions with no pawns
113 if (piece_count == 0) return;
114
115 // Pre-calculate direction and ranks once
116 const int direction = (us == Color::White) ? NORTH : SOUTH;
117 const Rank start_rank = (us == Color::White) ? Rank::R2 : Rank::R7;
118 // const Rank promo_rank = (us == Color::White) ? Rank::R8 : Rank::R1; // Not used - using promo_min/max instead
119
120 // Pre-calculate promotion rank bounds for fast comparison
121 const int promo_min = (us == Color::White) ? 91 : 21;
122 const int promo_max = (us == Color::White) ? 98 : 28;
123
124 for (int i = 0; i < piece_count; ++i) {
125 const int from = pos.pList[int(us)][int(PieceType::Pawn)][i];
126 if (from == -1) continue;
127
128 // Single push
129 const int to = from + direction;
130 if (IS_PLAYABLE(to) && pos.at(to) == Piece::None) {
131 // Fast promotion check using pre-calculated bounds
132 if (to >= promo_min && to <= promo_max) {
133 // Use optimized batch generation
134 generate_promotion_batch(list, from, to);
135 } else {
136 list.add_quiet_move(make_move(from, to));
137
138 // Double push from starting position
139 if (rank_of(from) == start_rank) {
140 const int double_to = to + direction;
141 if (IS_PLAYABLE(double_to) && pos.at(double_to) == Piece::None) {
142 list.add_quiet_move(make_pawn_start(from, double_to));
143 }
144 }
145 }
146 }
147
148 // Captures (diagonals) - unrolled for better performance
149 const int capture_left = from + direction + WEST;
150 const int capture_right = from + direction + EAST;
151
152 // Left capture
153 if (IS_PLAYABLE(capture_left)) {
154 const Piece target = pos.at(capture_left);
155 if (target != Piece::None && color_of(target) == !us) {
156 // Fast promotion check
157 if (capture_left >= promo_min && capture_left <= promo_max) {
158 generate_promotion_batch(list, from, capture_left, type_of(target));
159 } else {
160 list.add_capture_move(make_capture(from, capture_left, type_of(target)), pos);
161 }
162 } else if (capture_left == pos.ep_square) {
163 list.add_en_passant_move(make_en_passant(from, capture_left));
164 }
165 }
166
167 // Right capture
168 if (IS_PLAYABLE(capture_right)) {
169 const Piece target = pos.at(capture_right);
170 if (target != Piece::None && color_of(target) == !us) {
171 // Fast promotion check
172 if (capture_right >= promo_min && capture_right <= promo_max) {
173 generate_promotion_batch(list, from, capture_right, type_of(target));
174 } else {
175 list.add_capture_move(make_capture(from, capture_right, type_of(target)), pos);
176 }
177 } else if (capture_right == pos.ep_square) {
178 list.add_en_passant_move(make_en_passant(from, capture_right));
179 }
180 }
181 }
182}
