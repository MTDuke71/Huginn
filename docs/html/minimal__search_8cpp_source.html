244

246 if (white_rooks > 0 || black_rooks > 0 || white_queens > 0 || black_queens > 0) {
247 return false;
248 }
249
250 // Only minor pieces remain (bishops and knights)
251 int white_pieces = white_bishops + white_knights;
252 int black_pieces = black_bishops + black_knights;

255 // 1. K vs K

257 return true;

266 // 4. K+N vs K+N is generally drawn but can have winning positions

269 return false;

400 // Convert engine score to proper UCI format
401 // UCI specification:
402 // - cp <x>: score from engine's point of view in centipawns
403 // - mate <y>: mate in y MOVES (not plies). If engine is getting mated, use negative y
404
405 // The search returns scores from the side-to-move's perspective:
406 // - Positive score means side-to-move is winning
407 // - Negative score means side-to-move is losing
408
409 // MATE = 29000, so scores close to +/-MATE are mate scores
410 if (score > MATE - 100) {
411 // Positive mate score: side_to_move (engine) is mating opponent
412 int mate_in_plies = MATE - score;
413 int mate_in_moves = (mate_in_plies + 1) / 2; // Convert plies to moves

415 } else if (score < -MATE + 100) {
416 // Negative mate score: side_to_move (engine) is being mated
417 int mate_in_plies = MATE + score; // score is negative, so this is MATE - abs(score)
418 int mate_in_moves = (mate_in_plies + 1) / 2; // Convert plies to moves
419 return "mate -" + std::to_string(mate_in_moves); // Negative because engine is being mated
420 } else {
421 // Regular centipawn score from engine's perspective
422 return "cp " + std::to_string(score);

634 0 // King (should never be captured)

637 // Initialize MVV-LVA scores
638 // Higher scores = better captures to search first
639 // Formula: (victim_value * 100) + (600 - attacker_value)
640 // This prioritizes: valuable victims + cheap attackers
641 for (int victim = 0; victim < 7; victim++) {
642 for (int attacker = 0; attacker < 7; attacker++) {

644 // No victim = not a capture

648 // Example: Pawn(100) takes Queen(1000) = (1000 * 100) + (600 - 100) = 100,500
649 // Example: Queen(1000) takes Pawn(100) = (100 * 100) + (600 - 1000) = 9,600
