35 inline void generate_sliding_ray(const Position& pos, S_MOVELIST& list, Color us, int from) {
36 for (int to = from + Direction; IS_PLAYABLE(to); to += Direction) {
37 Piece target = pos.at(to);
38
39 if (target == Piece::None) {
