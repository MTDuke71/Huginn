102 explicit TranspositionTable(size_t size_mb = 64) {
103 // Calculate number of entries for given size in MB
104 size_t num_entries = (size_mb * 1024 * 1024) / sizeof(TTEntry);

106 // Round down to nearest power of 2 for fast indexing
107 size_t power_of_2 = 1;
108 while (power_of_2 * 2 <= num_entries) {
109 power_of_2 *= 2;
110 }

112 table.resize(power_of_2);
113 size_mask = power_of_2 - 1;
114 }

116 // Store position in transposition table with depth-preferred replacement
117 void store(uint64_t zobrist_key, int score, uint8_t depth, uint8_t node_type, uint32_t best_move = 0) {
118 size_t index = zobrist_key & size_mask;
119 TTEntry& entry = table[index];
120
121 // Depth-preferred replacement strategy:
122 // Replace if: 1) Empty slot, 2) Same position, 3) New search is deeper, or 4) Significantly older entry
123 bool should_replace = false;
124
125 if (entry.zobrist_key == 0) {
126 // Empty slot - always replace
127 should_replace = true;
128 } else if (entry.zobrist_key == zobrist_key) {
129 // Same position - always update with new information
130 should_replace = true;
131 } else if (depth >= entry.depth) {
132 // New search is deeper or equal - prefer deeper searches
133 should_replace = true;
134 } else {
135 // Keep existing deeper entry, but could add aging logic here in future
136 should_replace = false;
137 }
138
139 if (should_replace) {
140 entry.zobrist_key = zobrist_key;
141 entry.score = score;
142 entry.depth = depth;
143 entry.node_type = node_type;
144 entry.best_move = best_move;
145 writes++; // Track write operations
146 }
147 }
