66 void initialize_knight_tables() {
68 const int knight_deltas[8][2] = {
69 {-2, -1}, {-2, +1}, {-1, -2}, {-1, +2},
70 {+1, -2}, {+1, +2}, {+2, -1}, {+2, +1}
71 };
72
73 // Initialize lookup tables for all 64 squares
74 for (int sq64 = 0; sq64 < 64; ++sq64) {
75 int rank = square_to_rank(sq64);
76 int file = square_to_file(sq64);
77
78 int move_count = 0;
79 uint64_t attack_bitboard = 0ULL;
80
81 // Check all 8 possible knight moves from current square
82 for (int i = 0; i < 8; ++i) {
83 int new_rank = rank + knight_deltas[i][0];
84 int new_file = file + knight_deltas[i][1];
85
86 int target_sq = rank_file_to_square(new_rank, new_file);
87
88 if (target_sq != -1) { // Valid square (within board boundaries)
89 KNIGHT_MOVES[sq64][move_count] = target_sq;
90 attack_bitboard |= (1ULL << target_sq);
91 move_count++;
92 }
93 }
94
95 // Store computed values in lookup tables
96 KNIGHT_MOVE_COUNT[sq64] = move_count;
97 KNIGHT_ATTACKS[sq64] = attack_bitboard;
98
99 // Fill remaining array slots with -1 for safety and debugging
100 for (int i = move_count; i < 8; ++i) {
101 KNIGHT_MOVES[sq64][i] = -1;
102 }
103 }
104 }
