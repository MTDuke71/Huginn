77// ---------- Castling Rights ----------
78// Bit flags for castling permissions (can be combined with |)
79constexpr uint8_t CASTLE_NONE = 0; // No castling rights
80constexpr uint8_t CASTLE_WK = 1; // White King-side (WKCA)
81constexpr uint8_t CASTLE_WQ = 2; // White Queen-side (WQCA)
82constexpr uint8_t CASTLE_BK = 4; // Black King-side (BKCA)
83constexpr uint8_t CASTLE_BQ = 8; // Black Queen-side (BQCA)
84constexpr uint8_t CASTLE_ALL = 15; // All castling rights (KQkq)
85
86// Legacy aliases for compatibility
87constexpr int WKCA = CASTLE_WK;
88constexpr int WQCA = CASTLE_WQ;
89constexpr int BKCA = CASTLE_BK;
90constexpr int BQCA = CASTLE_BQ;

92// Castling utility functions
93constexpr inline bool can_castle_kingside(uint8_t rights, Color c) {
94 return c == Color::White ? (rights & CASTLE_WK) : (rights & CASTLE_BK);
95}
96
97constexpr inline bool can_castle_queenside(uint8_t rights, Color c) {
98 return c == Color::White ? (rights & CASTLE_WQ) : (rights & CASTLE_BQ);
99}
100
101constexpr inline bool can_castle(uint8_t rights, Color c) {
102 return can_castle_kingside(rights, c) || can_castle_queenside(rights, c);
103}

105constexpr inline uint8_t remove_castling_rights(uint8_t rights, Color c) {
106 return c == Color::White ? (rights & ~(CASTLE_WK | CASTLE_WQ))
107 : (rights & ~(CASTLE_BK | CASTLE_BQ));
108}

110// Helper function to update castling rights efficiently
111constexpr inline uint8_t update_castling_rights_for_squares(uint8_t current_rights, int from_square, int to_square) {
112 // Use bitwise operations to efficiently clear castling rights based on square involvement
113 uint8_t new_rights = current_rights;
114
115 // Clear castling rights when pieces move from or to key squares
116 // A1 (21): Clear White Queenside
117 if (from_square == 21 || to_square == 21) {
118 new_rights &= ~CASTLE_WQ;
119 }
120 // E1 (25): Clear both White castling rights (king move)
121 if (from_square == 25 || to_square == 25) {
122 new_rights &= ~(CASTLE_WK | CASTLE_WQ);
123 }
124 // H1 (28): Clear White Kingside
125 if (from_square == 28 || to_square == 28) {
126 new_rights &= ~CASTLE_WK;
127 }
128 // A8 (91): Clear Black Queenside
129 if (from_square == 91 || to_square == 91) {
130 new_rights &= ~CASTLE_BQ;
131 }
132 // E8 (95): Clear both Black castling rights (king move)
133 if (from_square == 95 || to_square == 95) {
134 new_rights &= ~(CASTLE_BK | CASTLE_BQ);
135 }
136 // H8 (98): Clear Black Kingside
137 if (from_square == 98 || to_square == 98) {
138 new_rights &= ~CASTLE_BK;
139 }
140
141 return new_rights;
142}

238// ---------- Castling Rights Optimization ----------
239// Lookup table for efficient castling rights updates during move making
240// Each square maps to a mask that preserves only the castling rights that should remain
241// when a piece moves from or to that square
242namespace CastlingLookup {
243 // Initialize castling permission lookup table
244 constexpr std::array<uint8_t, 120> create_castling_mask_table() {
245 std::array<uint8_t, 120> table{};
246
247 // Default: all castling rights preserved (15 = 0b1111)
248 for (size_t i = 0; i < 120; ++i) {
249 table[i] = CASTLE_WK | CASTLE_WQ | CASTLE_BK | CASTLE_BQ;
250 }
251
252 // Critical squares that affect castling rights:
253 table[21] = CASTLE_WK | CASTLE_BK | CASTLE_BQ; // a1 - clear white queenside
254 table[25] = CASTLE_BK | CASTLE_BQ; // e1 - clear both white rights
255 table[28] = CASTLE_WQ | CASTLE_BK | CASTLE_BQ; // h1 - clear white kingside
256 table[91] = CASTLE_WK | CASTLE_WQ | CASTLE_BK; // a8 - clear black queenside
257 table[95] = CASTLE_WK | CASTLE_WQ; // e8 - clear both black rights
258 table[98] = CASTLE_WK | CASTLE_WQ | CASTLE_BQ; // h8 - clear black kingside
259
260 return table;
261 }
262
263 // Compile-time generated lookup table
264 constexpr auto CASTLING_MASK = create_castling_mask_table();
265
266 // Fast castling rights update: AND with masks for both from and to squares
267 constexpr inline uint8_t update_castling_rights(uint8_t current_rights, int from_sq, int to_sq) {
268 return current_rights & CASTLING_MASK[from_sq] & CASTLING_MASK[to_sq];
269 }
270}
