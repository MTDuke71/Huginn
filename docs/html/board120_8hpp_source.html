94 constexpr FileRankLookups() : files{}, ranks{}, playable{} {
95 // Initialize all squares to None/false (offboard)
96 for (int i = 0; i < 120; ++i) {
97 files[i] = File::None;
98 ranks[i] = Rank::None;
99 playable[i] = false;

101 // Fill playable squares

103 for (int f = 0; f < 8; ++f) {

105 files[size_t(sq120)] = File(f);

109 }
110 }
