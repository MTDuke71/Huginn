62bool input_is_waiting() {
63#ifdef _WIN32
64 // Windows implementation using _kbhit()
65 return _kbhit() != 0;
66#else
67 // Unix/Linux implementation using select()
68 fd_set readfds;
69 struct timeval timeout;
70
71 FD_ZERO(&readfds);
72 FD_SET(STDIN_FILENO, &readfds);
73
74 timeout.tv_sec = 0;
75 timeout.tv_usec = 0;
76
77 return select(STDIN_FILENO + 1, &readfds, nullptr, nullptr, &timeout) > 0;
78#endif
79}
