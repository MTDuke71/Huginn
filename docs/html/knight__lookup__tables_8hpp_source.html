147 inline void generate_knight_moves_lookup(const Position& pos, S_MOVELIST& list, Color us) {
148 int piece_count = pos.pCount[int(us)][int(PieceType::Knight)];
149
150 // Early exit if no knights
151 if (piece_count == 0) return;
152
153 const auto& knight_list = pos.pList[int(us)][int(PieceType::Knight)];
154
155 for (int i = 0; i < piece_count; ++i) {
156 int from = knight_list[i];
157 if (from == -1) continue;
158
159 // Convert from 120-square to 64-square index
160 int from_64 = MAILBOX_MAPS.to64[from];
161 if (from_64 == -1) continue; // Invalid square
162
163 // Use lookup table - no boundary checking needed
164 int move_count = KNIGHT_MOVE_COUNT[from_64];
165 const int* moves = KNIGHT_MOVES[from_64];
166
167 for (int j = 0; j < move_count; ++j) {
168 int to_64 = moves[j];
169 int to = MAILBOX_MAPS.to120[to_64]; // Convert back to 120-square system
170
171 Piece target = pos.at(to);
172 if (target == Piece::None) {
173 list.add_quiet_move(make_move(from, to));
174 } else if (color_of(target) == !us) {
175 list.add_capture_move(make_capture(from, to, type_of(target)), pos);
176 }
177 }
178 }
179 }

210 if (from == -1) continue;

213 if (from_64 == -1) continue;

216 uint64_t attacks = KNIGHT_ATTACKS[from_64];

218 // Iterate through attacked squares using bit manipulation
219 while (attacks) {
