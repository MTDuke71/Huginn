31// Bitmasks for move encoding/decoding - extract specific bit ranges
32constexpr int MOVE_FROM_MASK = 0x0000007F; // Bits 0-6: source square
33constexpr int MOVE_TO_MASK = 0x00003F80; // Bits 7-13: destination square
34constexpr int MOVE_CAPTURED_MASK = 0x0003C000; // Bits 14-17: captured piece type
35constexpr int MOVE_ENPASSANT = 0x00040000; // Bit 18: en passant flag
36constexpr int MOVE_PAWNSTART = 0x00080000; // Bit 19: pawn double-push flag
37constexpr int MOVE_PROMOTED_MASK = 0x00F00000; // Bits 20-23: promoted piece type
38constexpr int MOVE_CASTLE = 0x01000000; // Bit 24: castle move flag

40// Bit shift positions for encoding operations
41constexpr int MOVE_FROM_SHIFT = 0; // No shift needed for bits 0-6
42constexpr int MOVE_TO_SHIFT = 7; // Shift destination to bits 7-13
43constexpr int MOVE_CAPTURED_SHIFT = 14; // Shift captured type to bits 14-17
44constexpr int MOVE_ENPASSANT_SHIFT = 18; // Shift en passant flag to bit 18
45constexpr int MOVE_PAWNSTART_SHIFT = 19; // Shift pawn start flag to bit 19
46constexpr int MOVE_PROMOTED_SHIFT = 20; // Shift promoted type to bits 20-23
47constexpr int MOVE_CASTLE_SHIFT = 24; // Shift castle flag to bit 24
