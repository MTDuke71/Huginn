32// (Piece position changes are handled by atomic operations)
33void Position::update_zobrist_for_move(const S_MOVE& m, Piece moving, Piece captured, uint8_t old_castling_rights, int old_ep_square) {
34 // XOR side to move flag to match full computation logic
35 // Full computation: if (side_to_move == Color::Black) key ^= Side;
36 // Since we always flip sides on a move, we always need to XOR the Side bit
37 zobrist_key ^= Zobrist::Side;
38
39 // XOR out old castling rights and XOR in new castling rights
40 zobrist_key ^= Zobrist::Castle[old_castling_rights & 0xF];
41 zobrist_key ^= Zobrist::Castle[castling_rights & 0xF];
42
43 // XOR out old en passant file (if any)
44 if (old_ep_square != -1) {
45 int old_ep_file = static_cast<int>(file_of(old_ep_square));
46 if (old_ep_file >= 0 && old_ep_file < 8) {
47 zobrist_key ^= Zobrist::EpFile[old_ep_file];
48 }
49 }
50 // XOR in new en passant file (if any)
51 if (ep_square != -1) {
52 int new_ep_file = static_cast<int>(file_of(ep_square));
53 if (new_ep_file >= 0 && new_ep_file < 8) {
54 zobrist_key ^= Zobrist::EpFile[new_ep_file];
55 }
56 }

101bool Position::set_from_fen(const std::string& fen) {
102 reset();
103 std::vector<std::string> tokens;
104 std::istringstream iss(fen);
105 std::string token;
106 while (iss >> token) {
107 tokens.push_back(token);
108 }
109 if (tokens.size() != 6) {
110 return false;
111 }
112 // Piece placement: basic validation
113 const std::string& placement = tokens[0];
114 int rank = 7;
115 int file = 0;
116 for (char ch : placement) {
117 if (ch == '/') {
118 if (file != 8) return false;
119 rank--;
120 file = 0;
121 } else if (isdigit(ch)) {
122 file += ch - '0';
123 } else if (from_char(ch) == Piece::None) {
124 return false;
125 } else {
126 int square = sq(static_cast<File>(file), static_cast<Rank>(rank));
127 board[square] = from_char(ch);
128 file++;
129 }
130 }
131 if (rank != 0 || file != 8) return false;
132
133 // Side to move: must be "w" or "b"
134 if (tokens[1] == "w") {
135 side_to_move = Color::White;
136 } else if (tokens[1] == "b") {
137 side_to_move = Color::Black;
138 } else {
139 return false;
140 }
141
142 // Castling rights: must be subset of "KQkq" or "-"
143 castling_rights = 0;
144 if (tokens[2] == "-") {
145 // No rights
146 } else {
147 for (char c : tokens[2]) {
148 if (c == 'K') castling_rights |= CASTLE_WK;
149 else if (c == 'Q') castling_rights |= CASTLE_WQ;
150 else if (c == 'k') castling_rights |= CASTLE_BK;
151 else if (c == 'q') castling_rights |= CASTLE_BQ;
152 else return false;
153 }
154 }
155
156 // En passant: must be "-" or valid square a3/a6/h3/h6 etc.
157 if (tokens[3] == "-") {
158 ep_square = -1;
159 } else if (tokens[3].size() == 2 && tokens[3][0] >= 'a' && tokens[3][0] <= 'h' && (tokens[3][1] == '3' || tokens[3][1] == '6')) {
160 File file = File(tokens[3][0] - 'a');
161 Rank rank = Rank(tokens[3][1] - '1');
162 ep_square = sq(file, rank);
163 } else {
164 return false;
165 }
166
167 // Move counters: must be valid integers
168 try {
169 halfmove_clock = std::stoi(tokens[4]);
170 fullmove_number = std::stoi(tokens[5]);
171 } catch (...) {
172 return false;
173 }
174
175 rebuild_counts();
176 update_zobrist_key();
177 return true;
178}

180std::string Position::to_fen() const {
181 std::string fen;
182 for (int rank = 7; rank >= 0; --rank) {
183 int empty_count = 0;
184 for (int file = 0; file < 8; ++file) {
185 int sq120 = sq(static_cast<File>(file), static_cast<Rank>(rank));
186 Piece piece = board[sq120];
187 if (is_none(piece)) {
188 empty_count++;
189 } else {
190 if (empty_count > 0) {
191 fen += std::to_string(empty_count);
192 empty_count = 0;
193 }
194 fen += to_char(piece);
195 }
196 }
197 if (empty_count > 0) {
198 fen += std::to_string(empty_count);
199 }
200 if (rank > 0) {
201 fen += '/';
202 }
203 }
204 fen += ' ';
205 fen += (side_to_move == Color::White) ? 'w' : 'b';
206 fen += ' ';
207 std::string castling;
208 if (castling_rights & CASTLE_WK) castling += 'K';
209 if (castling_rights & CASTLE_WQ) castling += 'Q';
210 if (castling_rights & CASTLE_BK) castling += 'k';
211 if (castling_rights & CASTLE_BQ) castling += 'q';
212 fen += castling.empty() ? std::string(1, '-') : castling;
213 fen += ' ';
214 if (ep_square == -1) {
215 fen += '-';
216 } else {
217 File file = file_of(ep_square);
218 Rank rank = rank_of(ep_square);
219 fen += char('a' + int(file));
220 fen += char('1' + int(rank));
221 }
222 fen += ' ';
223 fen += std::to_string(halfmove_clock);
224 fen += ' ';
225 fen += std::to_string(fullmove_number);

239void Position::rebuild_counts() {
240 // Clear all piece lists and counts
241 for (int color = 0; color < 2; ++color) {
242 for (int type = 0; type < int(PieceType::_Count); ++type) {
243 pCount[color][type] = 0;
244 for (int i = 0; i < MAX_PIECES_PER_TYPE; ++i) {
245 pList[color][type][i] = -1;
246 }
247 }
248 }
249 piece_counts.fill(0);
250 material_score[0] = 0;
251 material_score[1] = 0;
252
253 // Clear legacy pawn bitboards
254 pawns_bb[0] = 0ULL;
255 pawns_bb[1] = 0ULL;
256 all_pawns_bb = 0ULL;
257
258 // Clear new full bitboard structures
259 for (int color = 0; color < 2; ++color) {
260 color_bitboards[color] = 0ULL;
261 for (int type = 0; type < int(PieceType::_Count); ++type) {
262 piece_bitboards[color][type] = 0ULL;
263 }
264 }
265 occupied_bitboard = 0ULL;
266
267 king_sq[0] = -1;
268 king_sq[1] = -1;
269
270 // Scan board and update all derived state
271 for (int sq120 = 0; sq120 < 120; ++sq120) {
272 Piece piece = board[sq120];
273 if (!is_playable(sq120) || is_none(piece) || piece == Piece::Offboard) continue;
274 Color color = color_of(piece);
275 PieceType type = type_of(piece);
276 int color_idx = int(color);
277 int type_idx = int(type);
278 int idx = pCount[color_idx][type_idx];
279 pList[color_idx][type_idx][idx] = sq120;
280 pCount[color_idx][type_idx]++;
281 piece_counts[type_idx]++;
282 if (type == PieceType::King) {
283 king_sq[color_idx] = sq120;
284 }
285
286 // Update all bitboard representations
287 int s64 = MAILBOX_MAPS.to64[sq120];
288 if (s64 >= 0) {
289 // Update new full bitboard system
290 setBit(piece_bitboards[color_idx][type_idx], s64);
291 setBit(color_bitboards[color_idx], s64);
292 setBit(occupied_bitboard, s64);
293
294 // Update legacy pawn bitboards for compatibility
295 if (type == PieceType::Pawn) {
296 pawns_bb[color_idx] |= (1ULL << s64);
297 all_pawns_bb |= (1ULL << s64);
298 }
299 }
300 if (type != PieceType::King) {
301 material_score[color_idx] += value_of(piece);
302 }
303 }
304}

314int Position::MakeMove(const S_MOVE& move) {
315 // Debug assertions for move validity
316 DEBUG_ASSERT(is_playable(move.get_from()), "Move source square must be playable");
317 DEBUG_ASSERT(is_playable(move.get_to()), "Move destination square must be playable");
318
319 // Store history information for potential undo
320 int from = move.get_from();
321 int to = move.get_to();
322
323 // Store history before making the move (BEFORE any modifications)
324 if (ply >= static_cast<int>(move_history.size())) {
325 move_history.resize(ply + 1);
326 }
327
328 S_UNDO& undo = move_history[ply];
329 undo.move = move;
330 undo.castling_rights = castling_rights;
331 undo.ep_square = ep_square;
332 undo.halfmove_clock = halfmove_clock;
333 undo.zobrist_key = zobrist_key; // Save ORIGINAL zobrist key (BEFORE modifications)
334
335 // Update castling rights based on move (from/to squares)
336 uint8_t new_castling_rights = CastlingLookup::update_castling_rights(undo.castling_rights, from, to);
337 castling_rights = new_castling_rights;
338
339 // Set captured piece - handle en passant special case
340 if (move.is_en_passant()) {
341 Color moving_color = color_of(at(from));
342 int captured_pawn_sq;
343 if (moving_color == Color::White) {
344 captured_pawn_sq = to + SOUTH; // Black pawn south of target
345 } else {
346 captured_pawn_sq = to + NORTH; // White pawn north of target
347 }
348
349 Piece piece_at_captured_sq = at(captured_pawn_sq);
350 if (is_none(piece_at_captured_sq)) {
351 // This is an invalid en passant move - there's no pawn to capture!
352 return 0; // Return illegal move
353 }
354
355 undo.captured = piece_at_captured_sq; // Capture the pawn that's actually being removed
356 } else {
357 undo.captured = at(to); // Normal capture
358 }
359
360 // Save additional state needed for undo
361 save_derived_state(undo);
362
363 // Handle en passant captures BEFORE updating ep_square
364 if (move.is_en_passant()) {
365 Color moving_color = color_of(at(from));
366 int captured_pawn_sq;
367 if (moving_color == Color::White) {
368 captured_pawn_sq = to + SOUTH; // Black pawn south of target
369 } else {
370 captured_pawn_sq = to + NORTH; // White pawn north of target
371 }
372
373 // Clear the captured pawn using atomic operation
374 clear_piece(captured_pawn_sq);
375 }
376
377 // Handle 50-move rule: reset on pawn move or capture
378 Piece moving_piece = at(from);
379 if (type_of(moving_piece) == PieceType::Pawn || !is_none(at(to))) {
380 halfmove_clock = 0;
381 } else {
382 ++halfmove_clock;
383 }
384
385 // Increment ply counter
386 ++ply;
387
388 // Reset en passant square
389 ep_square = -1;
390
391 // Set en passant square for pawn double moves
392 if (type_of(moving_piece) == PieceType::Pawn && !move.is_capture()) {
393 int from_rank = int(rank_of(from));
394 int to_rank = int(rank_of(to));
395
396 if (abs(to_rank - from_rank) == 2) { // Pawn moved two squares
397 int ep_rank = (from_rank + to_rank) / 2;
398 ep_square = sq(file_of(to), Rank(ep_rank));
399 }
400 }
401
402 // Make the actual move using atomic operations
403 if (move.is_promotion()) {
404 // Handle capture before promotion
405 if (!is_none(at(to))) {
406 clear_piece(to); // Remove captured piece
407 }
408 clear_piece(from); // Remove pawn
409 add_piece(to, make_piece(color_of(moving_piece), move.get_promoted())); // Add promoted piece
410 } else {
411 // Handle normal capture
412 if (!is_none(at(to))) {
413 clear_piece(to); // Remove captured piece first
414 }
415 move_piece(from, to); // Then move the piece
416 }
417
418 // Handle castling - move the rook
419 if (move.is_castle()) {
420 Color king_color = color_of(moving_piece);
421 int rook_from, rook_to;
422
423 if (king_color == Color::White) {
424 if (to == sq(File::G, Rank::R1)) { // White kingside
425 rook_from = sq(File::H, Rank::R1);
426 rook_to = sq(File::F, Rank::R1);
427 } else { // White queenside
428 rook_from = sq(File::A, Rank::R1);
429 rook_to = sq(File::D, Rank::R1);
430 }
431 } else {
432 if (to == sq(File::G, Rank::R8)) { // Black kingside
433 rook_from = sq(File::H, Rank::R8);
434 rook_to = sq(File::F, Rank::R8);
435 } else { // Black queenside
436 rook_from = sq(File::A, Rank::R8);
437 rook_to = sq(File::D, Rank::R8);
438 }
439 }
440
441 // Move the rook using atomic operation
442 move_piece(rook_from, rook_to);
443 }
444
445 // Update king square if king moved
446 if (type_of(moving_piece) == PieceType::King) {
447 king_sq[int(color_of(moving_piece))] = to;
448 }
449
450 // Change side to move
451 side_to_move = !side_to_move;
452 if (side_to_move == Color::White) {
453 ++fullmove_number;
454 }
455
456 // Update zobrist_key incrementally using XOR (much faster than recomputing)
457 update_zobrist_for_move(move, moving_piece, undo.captured, undo.castling_rights, undo.ep_square);
458
459 // Check if move left current player's king in check
460 // Note: side_to_move has already been flipped, so we check the previous side's king
461 Color previous_side = !side_to_move;
462 int king_square = king_sq[int(previous_side)];
463
464 // Re-enable legality check now that we've debugged the Zobrist issues
465 // Use SqAttacked directly to check if king is attacked by current side
466 if (SqAttacked(king_square, *this, side_to_move)) {
467 // Move is illegal - undo it
468 TakeMove();
469 return 0; // Illegal move
470 }
471
472 return 1; // Legal move
473}

476// Undoes the last move by retrieving information from the history array
477void Position::TakeMove() {
478 // Must have at least one move to undo
479 DEBUG_ASSERT(ply > 0, "Cannot take move when ply is 0");
480
481 // Decrement ply first to get the correct history index
482 --ply;
483
484 DEBUG_ASSERT(ply >= 0, "Invalid ply after decrement in TakeMove");
485 DEBUG_ASSERT(ply < static_cast<int>(move_history.size()), "Ply exceeds move_history size in TakeMove");
486 S_UNDO& undo = move_history[ply];
487 S_MOVE move = undo.move;
488
489 // Extract move information - for TakeMove, from/to are effectively reversed
490 int from = move.get_from();
491 int to = move.get_to();
492
493 // Restore position state from history (let forced recomputation handle zobrist)
494 castling_rights = undo.castling_rights;
495 ep_square = undo.ep_square;
496 halfmove_clock = undo.halfmove_clock; // NOTE: Don't restore zobrist_key here - let incremental updates handle it
497 // The atomic operations will restore piece positions incrementally
498 // We just need to restore side/castling/en passant incrementally
499
500 // Handle en passant undo - restore captured pawn
501 if (move.is_en_passant()) {
502 Color moving_color = color_of(at(to));
503 int captured_pawn_sq;
504 if (moving_color == Color::White) {
505 captured_pawn_sq = to + SOUTH; // Black pawn was south of target
506 } else {
507 captured_pawn_sq = to + NORTH; // White pawn was north of target
508 }
509
510 // Restore the captured pawn (stored in undo.captured)
511 add_piece(captured_pawn_sq, undo.captured);
512 }
513
514 // Handle castling undo - move rook back to original position
515 if (move.is_castle()) {
516 Color king_color = color_of(at(to));
517 int rook_from, rook_to;
518
519 if (king_color == Color::White) {
520 if (to == sq(File::G, Rank::R1)) { // White kingside
521 rook_from = sq(File::H, Rank::R1); // Original rook position
522 rook_to = sq(File::F, Rank::R1); // Current rook position
523 } else { // White queenside
524 rook_from = sq(File::A, Rank::R1); // Original rook position
525 rook_to = sq(File::D, Rank::R1); // Current rook position
526 }
527 } else {
528 if (to == sq(File::G, Rank::R8)) { // Black kingside
529 rook_from = sq(File::H, Rank::R8); // Original rook position
530 rook_to = sq(File::F, Rank::R8); // Current rook position
531 } else { // Black queenside
532 rook_from = sq(File::A, Rank::R8); // Original rook position
533 rook_to = sq(File::D, Rank::R8); // Current rook position
534 }
535 }
536
537 // Move rook back to original position
538 move_piece(rook_to, rook_from);
539 }
540
541 // Move piece back to original square BEFORE adding captured piece
542 if (move.is_promotion()) {
543 // For promotion, remove promoted piece and restore original pawn
544 clear_piece(to); // Remove promoted piece
545 Color moving_color = color_of(undo.captured) == Color::None ?
546 (side_to_move == Color::White ? Color::Black : Color::White) :
547 (color_of(undo.captured) == Color::White ? Color::Black : Color::White);
548 add_piece(from, make_piece(moving_color, PieceType::Pawn)); // Restore pawn
549 } else {
550 // Regular move - move piece back
551 DEBUG_ASSERT(is_playable(to), "Invalid 'to' square in TakeMove");
552 DEBUG_ASSERT(is_playable(from), "Invalid 'from' square in TakeMove");
553 DEBUG_ASSERT(!is_none(at(to)), "No piece to move back from 'to' square");
554 move_piece(to, from);
555 }
556
557 // Add captured piece back to destination square (if any)
558 if (!is_none(undo.captured) && !move.is_en_passant()) {
559 add_piece(to, undo.captured);
560 }
561
562 // Update king square if king moved
563 Piece moved_piece = at(from);
564 if (type_of(moved_piece) == PieceType::King) {
565 king_sq[int(color_of(moved_piece))] = from;
566 }
567
568 // Flip side to move back
569 side_to_move = !side_to_move;
570 if (side_to_move == Color::Black) {
571 --fullmove_number;
572 }
573
574 // Restore the original zobrist key directly from history
575 // The atomic operations automatically handle piece position changes via XOR
576 zobrist_key = undo.zobrist_key;
577}
