226bool PolyglotBook::load_book(const std::string& path) {
227 clear();
228 book_path = path;
229
230 std::ifstream file(path, std::ios::binary);
231 if (!file.is_open()) {
232 std::cerr << "Error: Cannot open Polyglot book file: " << path << std::endl;
233 return false;
234 }
235
236 // Get file size to calculate number of entries
237 file.seekg(0, std::ios::end);
238 size_t file_size = file.tellg();
239 file.seekg(0, std::ios::beg);
240
241 if (file_size % 16 != 0) {
242 std::cerr << "Error: Invalid Polyglot book file size (not multiple of 16 bytes)" << std::endl;
243 return false;
244 }
245
246 size_t num_entries = file_size / 16;
247 entries.reserve(num_entries);
248
249 // Read all entries
250 for (size_t i = 0; i < num_entries; ++i) {
251 PolyglotEntry entry;
252
253 // Read in big-endian format (Polyglot standard)
254 file.read(reinterpret_cast<char*>(&entry.key), 8);
255 file.read(reinterpret_cast<char*>(&entry.move), 2);
256 file.read(reinterpret_cast<char*>(&entry.weight), 2);
257 file.read(reinterpret_cast<char*>(&entry.learn), 4);
258
259 // Convert from big-endian to native endian
260 entry.key = __builtin_bswap64(entry.key);
261 entry.move = __builtin_bswap16(entry.move);
262 entry.weight = __builtin_bswap16(entry.weight);
263 entry.learn = __builtin_bswap32(entry.learn);
264
265 entries.push_back(entry);
266 }
267
268 // Sort entries by key for binary search
269 std::sort(entries.begin(), entries.end(), [](const PolyglotEntry& a, const PolyglotEntry& b) {
270 return a.key < b.key;
271 });
272
273 is_loaded = true;
274 std::cout << "Loaded Polyglot book: " << path << " (" << num_entries << " entries)" << std::endl;
275 return true;
276}

278uint64_t PolyglotBook::get_polyglot_key(const Position& pos) const {
279 uint64_t key = 0;
280
281 // XOR piece positions
282 for (int sq = 0; sq < 120; ++sq) {
283 if (!is_valid_square(sq)) continue;
284
285 Piece piece = pos.board[sq];
286 if (piece == Piece::None) continue;
287
288 // Convert to 64-square index (file=0..7, rank=0..7)
289 int file = sq % 10 - 1; // a=0, b=1, ..., h=7
290 int rank = sq / 10 - 2; // rank 1=0, rank 2=1, ..., rank 8=7
291
292 // Polyglot piece kind mapping:
293 // black pawn=0, white pawn=1, black knight=2, white knight=3, ..., white king=11
294 PieceType pt = get_piece_type(piece);
295 Color pc = get_piece_color(piece);
296
297 // Calculate kind_of_piece according to Polyglot spec
298 int kind_of_piece;
299 if (pc == Color::Black) {
300 kind_of_piece = (int(pt) - 1) * 2; // 0, 2, 4, 6, 8, 10 for black pieces
301 } else {
302 kind_of_piece = (int(pt) - 1) * 2 + 1; // 1, 3, 5, 7, 9, 11 for white pieces
303 }
304
305 // Calculate offset: offset_piece = 64 * kind_of_piece + 8 * row + file
306 int offset_piece = PolyglotKeys::OFFSET_PIECE + 64 * kind_of_piece + 8 * rank + file;
307 key ^= PolyglotKeys::RANDOM64[offset_piece];
308 }
309
310 // XOR castling rights using RANDOM64 offsets
311 // white can castle short=0, white can castle long=1, black can castle short=2, black can castle long=3
312 if (pos.castling_rights & CASTLE_WK) key ^= PolyglotKeys::RANDOM64[PolyglotKeys::OFFSET_CASTLE + 0]; // white short
313 if (pos.castling_rights & CASTLE_WQ) key ^= PolyglotKeys::RANDOM64[PolyglotKeys::OFFSET_CASTLE + 1]; // white long
314 if (pos.castling_rights & CASTLE_BK) key ^= PolyglotKeys::RANDOM64[PolyglotKeys::OFFSET_CASTLE + 2]; // black short
315 if (pos.castling_rights & CASTLE_BQ) key ^= PolyglotKeys::RANDOM64[PolyglotKeys::OFFSET_CASTLE + 3]; // black long
316
317 // XOR en passant file (only if there's a pawn next to the pushed pawn)
318 if (pos.ep_square != -1) {
319 int ep_file = pos.ep_square % 10 - 1;
320 if (ep_file >= 0 && ep_file < 8) {
321 // Check if there's actually a pawn that could capture en passant
322 bool can_capture_ep = false;
323 int ep_rank = pos.ep_square / 10 - 2;
324
325 // Check left and right of the en passant square for opposing pawns
326 if (ep_file > 0) { // Check left
327 int left_sq = (ep_rank + 2) * 10 + (ep_file - 1 + 1);
328 Piece left_piece = pos.board[left_sq];
329 if (get_piece_type(left_piece) == PieceType::Pawn &&
330 get_piece_color(left_piece) == pos.side_to_move) {
331 can_capture_ep = true;
332 }
333 }
334 if (ep_file < 7) { // Check right
335 int right_sq = (ep_rank + 2) * 10 + (ep_file + 1 + 1);
336 Piece right_piece = pos.board[right_sq];
337 if (get_piece_type(right_piece) == PieceType::Pawn &&
338 get_piece_color(right_piece) == pos.side_to_move) {
339 can_capture_ep = true;
340 }
341 }
342
343 if (can_capture_ep) {
344 key ^= PolyglotKeys::RANDOM64[PolyglotKeys::OFFSET_EP + ep_file];
345 }
346 }
347 }
348
349 // XOR side to move: If WHITE is to move, XOR the RandomTurn entry
350 if (pos.side_to_move == Color::White) {
351 key ^= PolyglotKeys::RANDOM64[PolyglotKeys::OFFSET_TURN];
352 }
353
354 return key;
355}

357std::vector<const PolyglotEntry*> PolyglotBook::find_entries(uint64_t key) const {
358 std::vector<const PolyglotEntry*> found_entries;
359
360 // Binary search for first entry with this key
361 auto it = std::lower_bound(entries.begin(), entries.end(), key,
362 [](const PolyglotEntry& entry, uint64_t search_key) {
363 return entry.key < search_key;
364 });
365
366 // Collect all entries with this key
367 while (it != entries.end() && it->key == key) {
368 found_entries.push_back(&(*it));
369 ++it;
370 }
371
372 return found_entries;
373}

375S_MOVE PolyglotBook::get_book_move(const Position& pos) const {
376 if (!is_loaded) {
377 return S_MOVE(); // Empty move
378 }
379
380 uint64_t key = get_polyglot_key(pos);
381 auto book_entries = find_entries(key);
382
383 if (book_entries.empty()) {
384 return S_MOVE(); // No book moves
385 }
386
387 // Calculate total weight
388 uint32_t total_weight = 0;
389 for (const auto* entry : book_entries) {
390 total_weight += entry->weight;
391 }
392
393 if (total_weight == 0) {
394 return S_MOVE(); // No valid moves
395 }
396
397 // Choose move based on weight (probabilistic)
398 static std::random_device rd;
399 static std::mt19937 gen(rd());
400 std::uniform_int_distribution<uint32_t> dist(0, total_weight - 1);
401 uint32_t chosen_weight = dist(gen);
402
403 uint32_t current_weight = 0;
404 for (const auto* entry : book_entries) {
405 current_weight += entry->weight;
406 if (current_weight > chosen_weight) {
407 return polyglot_to_move(entry->move, pos);
408 }
409 }
410
411 // Fallback to first move
412 return polyglot_to_move(book_entries[0]->move, pos);
413}

415S_MOVE PolyglotBook::polyglot_to_move(uint16_t poly_move, const Position& pos) const {
416 // Polyglot move format: ttttttffffffkkkk
417 // where f=from square (6 bits), t=to square (6 bits), k=promotion piece kind (4 bits)
418 // BUT: Testing shows the bit order might be reversed from documentation
419
420 int to_sq = poly_move & 0x3F; // Bottom 6 bits = to square
421 int from_sq = (poly_move >> 6) & 0x3F; // Next 6 bits = from square
422 int promotion = (poly_move >> 12) & 0xF; // Top 4 bits = promotion
423
424 // Convert from 64-square to 120-square format
425 // 64-square: file=0..7, rank=0..7, square = rank*8 + file
426 // 120-square: file=1..8, rank=2..9, square = rank*10 + file
427 int from_file = from_sq % 8;
428 int from_rank = from_sq / 8;
429 int from_120 = (from_rank + 2) * 10 + (from_file + 1);
430
431 int to_file = to_sq % 8;
432 int to_rank = to_sq / 8;
433 int to_120 = (to_rank + 2) * 10 + (to_file + 1);
434
435 // Determine captured piece
436 Piece captured = pos.board[to_120];
437 PieceType captured_type = (captured != Piece::None) ? get_piece_type(captured) : PieceType::None;
438
439 // Check for special moves
440 bool is_ep = false;
441 bool is_castle = false;
442 bool is_pawn_start = false;
443
444 Piece moving_piece = pos.board[from_120];
445
446 // Check for en passant
447 if (get_piece_type(moving_piece) == PieceType::Pawn && to_120 == pos.ep_square) {
448 is_ep = true;
449 captured_type = PieceType::Pawn; // Captured piece is pawn
450 }
451
452 // Check for castling according to Polyglot convention
453 // Polyglot represents castling as king-to-rook moves:
454 // white short: e1h1, white long: e1a1, black short: e8h8, black long: e8a8
455 if (get_piece_type(moving_piece) == PieceType::King) {
456 // Check if this matches Polyglot castling pattern
457 bool is_white_king_on_e1 = (from_file == 4 && from_rank == 0 && get_piece_color(moving_piece) == Color::White);
458 bool is_black_king_on_e8 = (from_file == 4 && from_rank == 7 && get_piece_color(moving_piece) == Color::Black);
459
460 if (is_white_king_on_e1) {
461 // White castling: e1h1 (short) or e1a1 (long)
462 if ((to_file == 7 && to_rank == 0) || (to_file == 0 && to_rank == 0)) {
463 is_castle = true;
464 // For castling moves, there's no captured piece on the target square
465 captured_type = PieceType::None;
466
467 // Convert Polyglot king-to-rook move to standard king-to-destination move
468 if (to_file == 7) {
469 // Short castling: king goes to g1 (file 6)
470 to_120 = (0 + 2) * 10 + (6 + 1); // g1 in 120-square format
471 } else {
472 // Long castling: king goes to c1 (file 2)
473 to_120 = (0 + 2) * 10 + (2 + 1); // c1 in 120-square format
474 }
475 }
476 } else if (is_black_king_on_e8) {
477 // Black castling: e8h8 (short) or e8a8 (long)
478 if ((to_file == 7 && to_rank == 7) || (to_file == 0 && to_rank == 7)) {
479 is_castle = true;
480 // For castling moves, there's no captured piece on the target square
481 captured_type = PieceType::None;
482
483 // Convert Polyglot king-to-rook move to standard king-to-destination move
484 if (to_file == 7) {
485 // Short castling: king goes to g8 (file 6)
486 to_120 = (7 + 2) * 10 + (6 + 1); // g8 in 120-square format
487 } else {
488 // Long castling: king goes to c8 (file 2)
489 to_120 = (7 + 2) * 10 + (2 + 1); // c8 in 120-square format
490 }
491 }
492 }
493 }
494
495 // Check for pawn double move
496 if (get_piece_type(moving_piece) == PieceType::Pawn && abs(to_rank - from_rank) == 2) {
497 is_pawn_start = true;
498 }
499
500 // Convert promotion piece
501 PieceType promoted_type = PieceType::None;
502 if (promotion > 0) {
503 // Polyglot promotion: 1=Knight, 2=Bishop, 3=Rook, 4=Queen
504 switch (promotion) {
505 case 1: promoted_type = PieceType::Knight; break;
506 case 2: promoted_type = PieceType::Bishop; break;
507 case 3: promoted_type = PieceType::Rook; break;
508 case 4: promoted_type = PieceType::Queen; break;
509 }
510 }
511
512 return S_MOVE(from_120, to_120, captured_type, is_ep, is_pawn_start, promoted_type, is_castle);
513}

515std::vector<std::pair<S_MOVE, uint16_t>> PolyglotBook::get_all_book_moves(const Position& pos) const {
516 std::vector<std::pair<S_MOVE, uint16_t>> book_moves;
517

527 book_moves.emplace_back(move, entry->weight);
528 }
529
530 return book_moves;
