152 // Update derived state incrementally for a move (much faster than rebuild_counts)
153 void update_derived_state_for_move(const S_MOVE& m, Piece moving, Piece captured) {
154 Color moving_color = color_of(moving);
155 PieceType moving_type = type_of(moving);
156
157 // Handle captured piece
158 if (!is_none(captured)) {
159 --piece_counts[size_t(type_of(captured))];
160
161 // Subtract captured piece value from opponent's material (exclude kings)
162 if (type_of(captured) != PieceType::King) {
163 Color captured_color = color_of(captured);
164 if (captured_color != Color::None) {
165 material_score[size_t(captured_color)] -= value_of(captured);
166 }
167 }
168
169 // Remove captured pawn from bitboard
170 if (type_of(captured) == PieceType::Pawn) {
171 int s64_to = MAILBOX_MAPS.to64[m.get_to()];
172 if (s64_to >= 0) {
173 Color captured_color = color_of(captured);
174 if (captured_color != Color::None) {
175 popBit(pawns_bb[size_t(captured_color)], s64_to);
176 popBit(all_pawns_bb, s64_to);
177 }
178 }
179 }
180 }
181
182 // Handle promotion
183 if (m.is_promotion()) {
184 // Remove pawn, add promoted piece
185 --piece_counts[size_t(PieceType::Pawn)];
186 ++piece_counts[size_t(m.get_promoted())];
187
188 // Update material score: remove pawn value, add promoted piece value
189 material_score[size_t(moving_color)] -= value_of(make_piece(moving_color, PieceType::Pawn));
190 material_score[size_t(moving_color)] += value_of(make_piece(moving_color, m.get_promoted()));
191
192 // Remove pawn from bitboard (promoted piece isn't a pawn)
193 int s64_to = MAILBOX_MAPS.to64[m.get_to()];
194 if (s64_to >= 0) {
195 popBit(pawns_bb[size_t(moving_color)], s64_to);
196 popBit(all_pawns_bb, s64_to);
197 }
198 } else {
199 // Regular move - update piece-specific derived state
200 if (moving_type == PieceType::Pawn) {
201 // Update pawn bitboard
202 int s64_from = MAILBOX_MAPS.to64[m.get_from()];
203 int s64_to = MAILBOX_MAPS.to64[m.get_to()];
204 if (s64_from >= 0 && s64_to >= 0) {
205 popBit(pawns_bb[size_t(moving_color)], s64_from);
206 setBit(pawns_bb[size_t(moving_color)], s64_to);
207 popBit(all_pawns_bb, s64_from);
208 setBit(all_pawns_bb, s64_to);
209 }
210 }
211
212 if (moving_type == PieceType::King) {
213 // Update king square
214 king_sq[size_t(moving_color)] = m.get_to();
215 }
216 }
217 }

283 // Atomic piece movement - follows VICE MovePiece pattern
284 // Moves a piece from one square to another, updating all necessary data structures
285 void move_piece(int from_square, int to_square) {
286 DEBUG_ASSERT(is_playable(from_square), "Invalid source square for piece move");
287 DEBUG_ASSERT(is_playable(to_square), "Invalid destination square for piece move");
288
289 Piece piece = at(from_square);
290 DEBUG_ASSERT(!is_none(piece), "Cannot move piece from empty square");
291 DEBUG_ASSERT(is_none(at(to_square)), "Cannot move piece to occupied square");
292
293 Color piece_color = color_of(piece);
294 PieceType piece_type = type_of(piece);
295
296 // 1. Hash piece out of from square and into to square
297 zobrist_key ^= Zobrist::Piece[int(type_of(piece)) + (color_of(piece) == Color::Black ? 6 : 0)][from_square];
298 zobrist_key ^= Zobrist::Piece[int(type_of(piece)) + (color_of(piece) == Color::Black ? 6 : 0)][to_square];
299
300 // 2. Update pieces array
301 set(from_square, Piece::None);
302 set(to_square, piece);
303
304 // 3. Update all bitboard representations
305 int from_sq64 = MAILBOX_MAPS.to64[from_square];
306 int to_sq64 = MAILBOX_MAPS.to64[to_square];
307 if (from_sq64 >= 0 && to_sq64 >= 0) {
308 // Update new full bitboard system
309 popBit(piece_bitboards[size_t(piece_color)][size_t(piece_type)], from_sq64);
310 popBit(color_bitboards[size_t(piece_color)], from_sq64);
311 popBit(occupied_bitboard, from_sq64);
312
313 setBit(piece_bitboards[size_t(piece_color)][size_t(piece_type)], to_sq64);
314 setBit(color_bitboards[size_t(piece_color)], to_sq64);
315 setBit(occupied_bitboard, to_sq64);
316
317 // Update legacy pawn bitboards for compatibility
318 if (piece_type == PieceType::Pawn) {
319 popBit(pawns_bb[size_t(piece_color)], from_sq64);
320 popBit(all_pawns_bb, from_sq64);
321 setBit(pawns_bb[size_t(piece_color)], to_sq64);
322 setBit(all_pawns_bb, to_sq64);
323 }
324 }
325
326 // 4. Update piece list (find piece and update its square)
327 int color_idx = int(piece_color);
328 int type_idx = int(piece_type);
329 for (int i = 0; i < pCount[color_idx][type_idx]; ++i) {
330 if (pList[color_idx][type_idx][i] == from_square) {
331 pList[color_idx][type_idx][i] = to_square;
332 return;
333 }
334 }
335
336 DEBUG_ASSERT(false, "Piece not found in piece list during move_piece");
337 }

339 // Atomic piece removal - consolidates all operations for better performance
340 // Follows the VICE ClearPiece pattern but maintains Huginn's C++ style
341 void clear_piece(int square) {
342 DEBUG_ASSERT(is_playable(square), "Cannot clear piece from invalid square");
343
344 Piece piece = at(square);
345 if (is_none(piece)) return; // Nothing to clear
346
347 DEBUG_ASSERT(!is_offboard(piece), "Cannot clear offboard piece");
348
349 Color piece_color = color_of(piece);
350 PieceType piece_type = type_of(piece);
351
352 // 1. Update zobrist hash (XOR out the piece using Huginn's zobrist system)
353 zobrist_key ^= Zobrist::Piece[int(type_of(piece)) + (color_of(piece) == Color::Black ? 6 : 0)][square];
354
355 // 2. Clear the board square
356 set(square, Piece::None);
357
358 // 3. Update material score (kings can never be captured in chess)
359 material_score[size_t(piece_color)] -= value_of(piece);
360
361 // 4. Update piece counters
362 --piece_counts[size_t(piece_type)];
363
364 // 5. Update all bitboard representations
365 int sq64 = MAILBOX_MAPS.to64[square];
366 if (sq64 >= 0) {
367 // Update new full bitboard system
368 popBit(piece_bitboards[size_t(piece_color)][size_t(piece_type)], sq64);
369 popBit(color_bitboards[size_t(piece_color)], sq64);
370 popBit(occupied_bitboard, sq64);
371
372 // Update legacy pawn bitboards for compatibility
373 if (piece_type == PieceType::Pawn) {
374 popBit(pawns_bb[size_t(piece_color)], sq64);
375 popBit(all_pawns_bb, sq64);
376 }
377 }
378
379 // 6. Remove from piece list (atomic operation)
380 int color_idx = int(piece_color);
381 int type_idx = int(piece_type);
382
383 DEBUG_ASSERT(pCount[color_idx][type_idx] > 0, "Piece count already zero");
384
385 // Find the piece in the list and remove it efficiently
386 for (int i = 0; i < pCount[color_idx][type_idx]; ++i) {
387 if (pList[color_idx][type_idx][i] == square) {
388 // Replace with last piece in list and decrement count
389 --pCount[color_idx][type_idx];
390 pList[color_idx][type_idx][i] = pList[color_idx][type_idx][pCount[color_idx][type_idx]];
391 pList[color_idx][type_idx][pCount[color_idx][type_idx]] = -1; // Clear old position
392 return;
393 }
394 }
395
396 DEBUG_ASSERT(false, "Piece not found in piece list during clear_piece");
397 }

447 // Enhanced move making with full undo support
448 void make_move_with_undo(const S_MOVE& m) {
449 // Debug assertions for move validity
450 DEBUG_ASSERT(is_playable(m.get_from()), "Move source square must be playable");
451 DEBUG_ASSERT(is_playable(m.get_to()), "Move destination square must be playable");
452
453 // Safety check: if source square is empty, skip this move (piece list corruption recovery)
454 if (is_none(at(m.get_from()))) {
455 return; // Silently ignore invalid moves to prevent crashes
456 }
457
458 // Ensure move_history has enough capacity
459 if (ply >= static_cast<int>(move_history.size())) {
460 move_history.resize(ply + 1);
461 }
462
463 S_UNDO& undo = move_history[ply]; // Vector access
464 undo.move = m; // Store the complete S_MOVE
465 undo.castling_rights = castling_rights;
466 undo.ep_square = ep_square;
467 undo.halfmove_clock = halfmove_clock;
468 undo.zobrist_key = zobrist_key;
469 undo.captured = at(m.get_to());
470
471 // Save derived state for efficient undo (performance optimization)
472 save_derived_state(undo);
473
474#ifdef DEBUG_CASTLING
475 std::cout << "[DEBUG] Before move: " << m.get_from() << "->" << m.get_to() << " castle? " << m.is_castle() << " rights: " << int(castling_rights) << std::endl;
476 std::cout << "[DEBUG] White King list: ";
477 for (int i = 0; i < pCount[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::King)]; ++i)
478 std::cout << pList[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::King)][i] << " ";
479 std::cout << " | White Rook list: ";
480 for (int i = 0; i < pCount[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::Rook)]; ++i)
481 std::cout << pList[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::Rook)][i] << " ";
482 std::cout << std::endl;
483#endif
484 // --- Efficient castling rights update using lookup table ---
485 // Single operation replaces multiple conditional checks for better performance
486 int from = m.get_from();
487 int to = m.get_to();
488
489 // Update castling rights using optimized lookup table (single AND operation)
490 castling_rights = CastlingLookup::update_castling_rights(castling_rights, from, to);
491
492#ifdef DEBUG_CASTLING
493 std::cout << "[DEBUG] After move: " << m.get_from() << "->" << m.get_to() << " rights: " << int(castling_rights) << std::endl;
494 std::cout << "[DEBUG] White King list: ";
495 for (int i = 0; i < pCount[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::King)]; ++i)
496 std::cout << pList[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::King)][i] << " ";
497 std::cout << " | White Rook list: ";
498 for (int i = 0; i < pCount[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::Rook)]; ++i)
499 std::cout << pList[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::Rook)][i] << " ";
500 std::cout << std::endl;
501#endif
502
503 // Make the move
504 Piece moving = at(m.get_from());
505 Piece captured = undo.captured;
506
507 if (type_of(moving) == PieceType::Pawn || !is_none(captured)) {
508 halfmove_clock = 0;
509 } else {
510 ++halfmove_clock;
511 }
512
513 // Update piece lists before changing the board
514 Color moving_color = color_of(moving);
515 PieceType moving_type = type_of(moving);
516
517 // Remove captured piece using atomic operation for better performance
518 if (!is_none(captured)) {
519 clear_piece(m.get_to());
520 }
521
522 // Handle en passant captures
523 if (m.is_en_passant()) {
524 // The captured pawn is not on the destination square, but adjacent to it
525 int captured_pawn_sq;
526 if (moving_color == Color::White) {
527 // White captures en passant - captured black pawn is south of target square
528 captured_pawn_sq = m.get_to() + SOUTH;
529 } else {
530 // Black captures en passant - captured white pawn is north of target square
531 captured_pawn_sq = m.get_to() + NORTH;
532 }
533
534 Piece captured_pawn = at(captured_pawn_sq);
535 if (!is_none(captured_pawn)) {
536 // Use atomic clear_piece for en passant capture
537 clear_piece(captured_pawn_sq);
538
539 // Update undo information to remember the captured pawn
540 undo.captured = captured_pawn;
541 }
542 }
543
544 // Handle promotion - use atomic operations for better performance
545 if (m.is_promotion()) {
546 clear_piece(m.get_from()); // Remove pawn
547 add_piece(m.get_to(), make_piece(moving_color, m.get_promoted())); // Add promoted piece
548 } else {
549 // Regular move - use atomic move_piece for better performance
550 move_piece(m.get_from(), m.get_to());
551 }
552
553 // Handle castling move - place rook correctly
554 if (m.is_castle()) {
555 Color king_color = color_of(moving);
556 int rook_from, rook_to;
557
558 if (king_color == Color::White) {
559 if (m.get_to() == sq(File::G, Rank::R1)) { // Kingside
560 rook_from = sq(File::H, Rank::R1);
561 rook_to = sq(File::F, Rank::R1);
562 } else { // Queenside
563 rook_from = sq(File::A, Rank::R1);
564 rook_to = sq(File::D, Rank::R1);
565 }
566 } else {
567 if (m.get_to() == sq(File::G, Rank::R8)) { // Kingside
568 rook_from = sq(File::H, Rank::R8);
569 rook_to = sq(File::F, Rank::R8);
570 } else { // Queenside
571 rook_from = sq(File::A, Rank::R8);
572 rook_to = sq(File::D, Rank::R8);
573 }
574 }
575
576 // Move the rook - but first verify it's actually there
577 Piece rook = at(rook_from);
578 if (is_none(rook) || type_of(rook) != PieceType::Rook || color_of(rook) != king_color) {
579#ifndef NDEBUG
580 std::cout << "CASTLING ERROR: Expected " << (king_color == Color::White ? "white" : "black")
581 << " rook at square " << rook_from << " but found piece " << static_cast<int>(rook) << std::endl;
582#endif
583 // This should not happen in a legal position - just skip the rook move
584 return;
585 }
586 // Use atomic move_piece for castling rook movement
587 move_piece(rook_from, rook_to);
588 }
589
590 ep_square = -1; // Reset, then check for pawn double moves
591
592 // Set en passant square for pawn double moves
593 if (type_of(moving) == PieceType::Pawn) {
594 int from_rank = int(rank_of(m.get_from()));
595 int to_rank = int(rank_of(m.get_to()));
596 int rank_diff = abs(to_rank - from_rank);
597
598 // Check if pawn moved two squares
599 if (rank_diff == 2) {
600 // Set en passant square to the square "behind" the pawn
601 int ep_rank = (from_rank + to_rank) / 2; // Square between from and to
602 File file = file_of(m.get_to());
603 ep_square = sq(file, Rank(ep_rank));
604 }
605 }
606
607 side_to_move = !side_to_move;
608 if (side_to_move == Color::White) ++fullmove_number;
609
610 // Update zobrist_key incrementally using XOR (much faster than recomputing)
611 update_zobrist_for_move(m, moving, captured, undo.castling_rights, undo.ep_square);
612
613 // Increment ply after successful move
614 ++ply;
615 }

617 // Undo the last move
618 bool undo_move() {
619 #ifdef DEBUG_CASTLING
620 std::cout << "[DEBUG] After undo move rights: " << int(castling_rights) << std::endl;
621 std::cout << "[DEBUG] White King list: ";
622 for (int i = 0; i < pCount[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::King)]; ++i)
623 std::cout << pList[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::King)][i] << " ";
624 std::cout << " | White Rook list: ";
625 for (int i = 0; i < pCount[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::Rook)]; ++i)
626 std::cout << pList[static_cast<size_t>(Color::White)][static_cast<size_t>(PieceType::Rook)][i] << " ";
627 std::cout << std::endl;
628 #endif
629 if (ply == 0) return false; // No moves to undo
630
631 // Decrement ply first to get the correct index
632 --ply;
633 S_UNDO& undo = move_history[ply]; // Vector access
634
635 // Extract move information using modern getter methods (30-34% faster than decode_move)
636 int from = undo.move.get_from();
637 int to = undo.move.get_to();
638 PieceType promo = undo.move.get_promoted();
639
640 // Undo side to move first
641 side_to_move = !side_to_move;
642 if (side_to_move == Color::Black) --fullmove_number;
643
644 // Get the piece that moved
645 Piece moved = at(to);
646
647 // Handle promotion undo - restore piece lists
648 if (promo != PieceType::None) {
649 // Remove promoted piece from lists and add pawn back
650 remove_piece_from_list(color_of(moved), type_of(moved), to);
651 moved = make_piece(color_of(moved), PieceType::Pawn);
652 add_piece_to_list(color_of(moved), PieceType::Pawn, from);
653 } else {
654 // Regular move - update piece location in lists
655 move_piece_in_list(color_of(moved), type_of(moved), to, from);
656 }
657
658 // Move piece back
659 set(from, moved);
660
661 // Handle castling undo - restore rook position
662 if (undo.move.is_castle()) {
663 Color king_color = color_of(moved);
664 int rook_from, rook_to;
665
666 if (king_color == Color::White) {
667 if (to == sq(File::G, Rank::R1)) { // Kingside
668 rook_from = sq(File::H, Rank::R1);
669 rook_to = sq(File::F, Rank::R1);
670 } else { // Queenside
671 rook_from = sq(File::A, Rank::R1);
672 rook_to = sq(File::D, Rank::R1);
673 }
674 } else {
675 if (to == sq(File::G, Rank::R8)) { // Kingside
676 rook_from = sq(File::H, Rank::R8);
677 rook_to = sq(File::F, Rank::R8);
678 } else { // Queenside
679 rook_from = sq(File::A, Rank::R8);
680 rook_to = sq(File::D, Rank::R8);
681 }
682 }
683
684 // Move the rook back to its original position
685 Piece rook = at(rook_to);
686 set(rook_from, rook);
687 set(rook_to, Piece::None);
688 move_piece_in_list(king_color, PieceType::Rook, rook_to, rook_from);
689 }
690
691 // Handle en passant undo
692 if (undo.move.is_en_passant()) {
693 // Restore the captured pawn to its original square
694 Color moving_color = color_of(moved);
695 int captured_pawn_sq;
696 if (moving_color == Color::White) {
697 // White captured en passant - restore black pawn south of destination
698 captured_pawn_sq = to + SOUTH;
699 } else {
700 // Black captured en passant - restore white pawn north of destination
701 captured_pawn_sq = to + NORTH;
702 }
703
704 // Restore the captured pawn and update piece lists
705 set(captured_pawn_sq, undo.captured);
706 set(to, Piece::None); // Clear the destination square
707
708 // Restore the captured pawn to piece lists
709 if (!is_none(undo.captured)) {
710 add_piece_to_list(color_of(undo.captured), PieceType::Pawn, captured_pawn_sq);
711 }
712 } else {
713 set(to, undo.captured); // Restore captured piece (or Piece::None)
714
715 // Restore captured piece to piece lists
716 if (!is_none(undo.captured)) {
717 add_piece_to_list(color_of(undo.captured), type_of(undo.captured), to);
718 }
719 }
720
721 // Restore position state
722 castling_rights = undo.castling_rights;
723 ep_square = undo.ep_square;
724 halfmove_clock = undo.halfmove_clock;
725 zobrist_key = undo.zobrist_key;
726
727 // Restore derived state incrementally (much faster than rebuild_counts)
728 restore_derived_state(undo);
729
730 return true;
731 }
